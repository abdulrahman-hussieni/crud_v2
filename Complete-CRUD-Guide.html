namespace crud_v2.DAL.Entities
{
    public class Employee
    {
        // Primary Key - Auto-incremented by database
        public int Id { get; set; }
        
        // Employee basic information
        public string Name { get; set; }
        public double Salary { get; set; }

        // Image filename stored in database (not the actual image)
        // Example: "a1b2c3d4-e5f6-7890-abcd-ef1234567890profile.jpg"
        public string? File { get; set; }

        // Audit fields - Track when records are created/updated/deleted
        public DateTime CreatedOn { get; set; } = DateTime.Now;
        public DateTime? UpdatedOn { get; set; }
        public DateTime? DeletedOn { get; set; }

        // Track who performed the action (optional - for future use)
        public string? CreatedBy { get; set; }
        public string? UpdatedBy { get; set; }
        public string? DeletedBy { get; set; }
    }
}
üí° Understanding the Entity:
  • Id - Automatically generated by database
  • Name & Salary - Core employee data
  • File - Stores only the filename (with ?) meaning it's nullable/optional
  • DateTime? - The ? means nullable (can be null)
  • CreatedOn - Automatically set to current time when created

Create Database Context

3 Create DAL/DB/ApplicationDbContext.cs: 
using crud_v2.DAL.Entities;
using Microsoft.EntityFrameworkCore;

namespace crud_v1.DAL.DB
{
    public class ApplicationDbContext : DbContext
    {
        // Configure database connection
        protected override void OnConfiguring(DbContextOptionsBuilder optionsBuilder)
        {
            optionsBuilder.UseSqlServer(
                "Server=.;Database=EMPDB;Trusted_Connection=True;TrustServerCertificate=True;"
            );
        }

        // Define database table
        // This creates a table named "employees" in the database
        public DbSet<Employee> employees { get; set; }
    }
}
üí° Understanding Connection String:
  • Server=. - Local SQL Server (dot means localhost)
  • Database=EMPDB - Database name (will be created)
  • Trusted_Connection=True - Use Windows Authentication
  • TrustServerCertificate=True - Required for local development

05. Create File Upload Helper Class

üéØ What we're doing: Creating a reusable helper class to handle file uploads and deletions.

File Upload Process Flow

Step 1: User selects file in browser
Step 2: Form submits with enctype="multipart/form-data"
Step 3: Controller receives IFormFile object
Step 4: Helper generates unique filename (GUID + original name)
Step 5: File saved to wwwroot/Files/ImageProfile/
Step 6: Only filename (not path) stored in database

Create Upload Helper Class

1 Create Helper/Upload.cs: 
namespace crud_v2.Helper
{
    public class Upload
    {
        /// <summary>
        /// Uploads a file to the specified folder and returns the saved filename
        /// </summary>
        /// <param name="FolderName">Subfolder name under wwwroot/Files/</param>
        /// <param name="File">The uploaded file</param>
        /// <returns>Generated filename or error message</returns>
        public static string UploadFile(string FolderName, IFormFile File)
        {
            try
            {
                // STEP 1: Build the folder path
                // Example: C:\MyProject\wwwroot\Files\ImageProfile\
                string FolderPath = Directory.GetCurrentDirectory() + "/wwwroot/Files/" + FolderName;

                // STEP 2: Create a unique filename
                // GUID generates a unique 36-character identifier
                // Example: "a1b2c3d4-e5f6-7890-abcd-ef1234567890profile.jpg"
                string FileName = Guid.NewGuid() + Path.GetFileName(File.FileName);

                // STEP 3: Combine folder path with filename
                // Example: C:\MyProject\wwwroot\Files\ImageProfile\a1b2c3d4...profile.jpg
                string FinalPath = Path.Combine(FolderPath, FileName);

                // STEP 4: Save the file to disk
                // Using 'using' ensures the file stream is properly closed
                using (var Stream = new FileStream(FinalPath, FileMode.Create))
                {
                    File.CopyTo(Stream);
                }

                // STEP 5: Return only the filename (not the full path)
                // We store only filename in database for portability
                return FileName;
            }
            catch (Exception ex)
            {
                // Return error message if something goes wrong
                return ex.Message;
            }
        }

        /// <summary>
        /// Deletes a file from the specified folder
        /// </summary>
        /// <param name="FolderName">Subfolder name under wwwroot/Files/</param>
        /// <param name="fileName">Name of the file to delete</param>
        /// <returns>Success or error message</returns>
        public static string RemoveFile(string FolderName, string fileName)
        {
            try
            {
                // Build the complete file path
                var directory = Path.Combine(
                    Directory.GetCurrentDirectory(), 
                    "wwwroot/Files", 
                    FolderName, 
                    fileName
                );

                // Check if file exists before trying to delete
                if (File.Exists(directory))
                {
                    File.Delete(directory);
                    return "File Deleted";
                }

                return "File Not Found";
            }
            catch (Exception ex)
            {
                return ex.Message;
            }
        }
    }
}

06. Create Employee Controller with CRUD Operations

üéØ What we're doing: Creating the controller that handles all employee operations - Create, Read, Update, Delete.

Create Controller File

1 Create Controllers/EmployeeController.cs: 
using crud_v1.DAL.DB;
using crud_v2.DAL.Entities;
using crud_v2.Helper;
using crud_v2.ModelVM.EmployeeVM;
using crud_v2.ViewModels;
using Microsoft.AspNetCore.Mvc;

namespace crud_v2.Controllers
{
    public class EmployeeController : Controller
    {
        // Create database context instance
        // This gives us access to the database
        private readonly ApplicationDbContext context = new ApplicationDbContext();

        // ==========================================
        // [1] READ - Show All Employees
        // ==========================================
        public IActionResult showAll()
        {
            // Query database and map to ViewModel
            var result = context.employees
                .Select(e => new ShowAllEmployeeVM
                {
                    Id = e.Id,
                    Name = e.Name,
                    Salary = e.Salary,
                    File = e.File,
                    CreatedOn = e.CreatedOn
                })
                .ToList();
            
            return View(result);
        }

        // ==========================================
        // [2] CREATE - GET (Show Create Form)
        // ==========================================
        public IActionResult Create()
        {
            // Just return the empty form
            return View();
        }

        // ==========================================
        // [2] CREATE - POST (Save New Employee)
        // ==========================================
        [HttpPost]
        public IActionResult SaveData(CreateEmployeeVM newemployee)
        {
            // STEP 1: Upload the image file
            string image = Upload.UploadFile("ImageProfile", newemployee.File);

            // STEP 2: Validate required fields (server-side validation)
            if (newemployee.Name == null || newemployee.Salary == 0)
            {
                return View(viewName: "Create");
            }

            // STEP 3: Map ViewModel to Entity
            Employee emp = new Employee()
            {
                Name = newemployee.Name,
                Salary = newemployee.Salary,
                File = image,  // Store the generated filename
                CreatedOn = DateTime.Now
            };

            // STEP 4: Add to database
            context.employees.Add(emp);

            // STEP 5: Save changes to database
            context.SaveChanges();

            // STEP 6: Redirect to the list page
            return RedirectToAction("showAll");
        }

        // ==========================================
        // [3] UPDATE - GET (Show Edit Form)
        // ==========================================
        public IActionResult Edit(int id)
        {
            // STEP 1: Find employee by ID
            var employee = context.employees.Find(id);

            // STEP 2: Check if employee exists
            if (employee == null)
            {
                return NotFound();
            }

            // STEP 3: Map Entity to ViewModel
            var model = new EditEmployeeVM
            {
                Id = employee.Id,
                Name = employee.Name,
                Salary = employee.Salary,
                ExistingFile = employee.File  // Keep current image filename
            };

            return View(model);
        }

        // ==========================================
        // [3] UPDATE - POST (Save Changes)
        // ==========================================
        [HttpPost]
        public IActionResult UpdateData(EditEmployeeVM editEmployee)
        {
            // STEP 1: Validate required fields
            if (editEmployee.Name == null || editEmployee.Salary == 0)
            {
                return View(viewName: "Edit", editEmployee);
            }

            // STEP 2: Find the existing employee in database
            var employee = context.employees.Find(editEmployee.Id);

            if (employee == null)
            {
                return NotFound();
            }

            // STEP 3: Update basic employee data
            employee.Name = editEmployee.Name;
            employee.Salary = editEmployee.Salary;
            employee.UpdatedOn = DateTime.Now;

            // STEP 4: Handle file upload (if user uploaded new image)
            if (editEmployee.File != null && editEmployee.File.Length > 0)
            {
                // Delete old file if it exists
                if (!string.IsNullOrEmpty(employee.File))
                {
                    Upload.RemoveFile("ImageProfile", employee.File);
                }

                // Upload new file
                employee.File = Upload.UploadFile("ImageProfile", editEmployee.File);
            }
            // If no new file uploaded, keep existing file (do nothing)

            // STEP 5: Save changes to database
            context.SaveChanges();

            // STEP 6: Redirect to list page
            return RedirectToAction("showAll");
        }

        // ==========================================
        // [4] DELETE - GET (Show Confirmation Page)
        // ==========================================
        public IActionResult Delete(int id)
        {
            // STEP 1: Find employee by ID
            var employee = context.employees.Find(id);

            // STEP 2: Check if employee exists
            if (employee == null)
            {
                return NotFound();
            }

            // STEP 3: Map Entity to ViewModel
            var model = new DeleteEmployeeVM
            {
                Id = employee.Id,
                Name = employee.Name,
                Salary = employee.Salary,
                File = employee.File,
                CreatedOn = employee.CreatedOn
            };

            return View(model);
        }

        // ==========================================
        // [4] DELETE - POST (Actually Delete)
        // ==========================================
        [HttpPost]
        public IActionResult ConfirmDelete(int id)
        {
            // STEP 1: Find the employee
            var employee = context.employees.Find(id);

            if (employee == null)
            {
                return NotFound();
            }

            // STEP 2: Delete the file if exists
            if (!string.IsNullOrEmpty(employee.File))
            {
                Upload.RemoveFile("ImageProfile", employee.File);
            }

            // STEP 3: Remove employee from database
            context.employees.Remove(employee);

            // STEP 4: Save changes
            context.SaveChanges();

            // STEP 5: Redirect to list page
            return RedirectToAction("showAll");
        }
    }
}
üí° Understanding HTTP Methods:
Method Usage Example
GET Retrieve data / Show forms Display edit form
POST Submit data / Make changes Save employee data
‚ö†Ô∏è Important Pattern: Each CRUD operation has TWO actions:
  • GET action - Shows the form/page
  • POST action - Processes the submitted data
